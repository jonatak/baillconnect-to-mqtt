#include <tunables/global>

profile baillconnect_to_mqtt flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Home Assistant's S6 startup scripts need the base file access expected by
  # the official add-on AppArmor profile template.
  file,

  # The bridge only needs outbound HTTP(S) to Baillconnect and MQTT to the broker.
  network inet stream,
  network inet6 stream,

  signal (send,receive) set=(kill,term,int,hup,cont),

  # Home Assistant base images start CMD through S6.
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  /usr/lib/bashio/** ix,
  /app/baillconnect-to-mqtt rix,
  /data/** rw,

  /etc/hosts r,
  /etc/nsswitch.conf r,
  /etc/resolv.conf r,
  /etc/ssl/certs/** r,
  /etc/localtime r,

  /tmp/** rwk,
}
